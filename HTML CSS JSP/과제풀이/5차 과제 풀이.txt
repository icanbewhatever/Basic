    <script src="js/jquery-3.7.1.min.js"></script>

    <script>
        //초기 인덱스 값
    
           
            var slideIndex = 1;
           

            $(function(){


                $('#prevButton').click(function(){
        
                    $('.slideItem').each(function(index, value){
                        // active 제거
                        $(this).removeClass('active');
                        ///console.log(index);
                        
                    });

                    if(slideIndex<=1) {
                        slideIndex=3;
                    }else{
                        slideIndex--;
                    }
                    console.log(slideIndex);

                    $('.slideItem').each(function(index, value){
                        if(index == slideIndex-1){
                            $(this).addClass('active');
                        }

                    });
                                        
                });
                 
                

                $('#nextButton').click(function(){
                    //다음 이미지 경고창 - alter('다음 이미지');
                  
                    // 이미지 영역 slideItem 모두 선택 $('.slideItem').length; 이미지 갯수
                    $('.slideItem').each(function(index, value){
                        // active 제거
                        $(this).removeClass('active');
                        //console.log(index);
                        
                    });

                    if(slideIndex>=3) {
                        slideIndex=1;
                        }else{
                            slideIndex++;
                        }
                    console.log(slideIndex);
                    //다음 누르면 slideIndex 증가
                  $('.slideItem').each(function(index, value){
                        if(index == slideIndex-1){
                            $(this).addClass('active');
                        }

                    });
                    
                });                 
              

            });
                
        
     

       
    </script>
