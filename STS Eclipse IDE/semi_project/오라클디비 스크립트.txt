   CREATE SEQUENCE ORDER_PRODUCT_SEQ
   START WITH 1
   INCREMENT BY 1
   NOMAXVALUE
   NOCYCLE;
   
CREATE TABLE ORDER_PRODUCT (
    order_num NUMBER DEFAULT ORDER_PRODUCT_SEQ.NEXTVAL PRIMARY KEY,
    order_date DATE DEFAULT SYSDATE,
    requester VARCHAR2(20) NOT NULL,
    brand_name VARCHAR2(20) NOT NULL,
    item_type VARCHAR2(20),
    order_place VARCHAR2(20) NOT NULL,
    item_name VARCHAR2(20) NOT NULL,
    item_size VARCHAR2(10) NOT NULL,
    quantity NUMBER NOT NULL,
    chest NUMBER,
    front_length NUMBER,
    sleeve_length NUMBER,
    back_width NUMBER,
    arm_width NUMBER,
    requirements VARCHAR2(200),
    product_details VARCHAR2(200)
    );
